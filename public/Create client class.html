import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';

export interface Delivery {
  id?: number;
  clientId?: number;
  chauffeurId?: number;
  status?: string;
  gouvernorat?: string;
  latitude?: number;
  longitude?: number;
  // ajoutez d'autres champs selon votre classe Delivery
}

@Injectable({
  providedIn: 'root'
})
export class DeliveryService {

  private apiUrl = 'http://localhost:8080/api/deliveries'; // changez le port si nécessaire

  constructor(private http: HttpClient) { }

  // Créer une livraison pour un client
  createDelivery(clientId: number, delivery: Delivery): Observable<Delivery> {
    return this.http.post<Delivery>(`${this.apiUrl}/${clientId}`, delivery);
  }

  // Récupérer les livraisons disponibles à proximité
  getAvailableDeliveries(gouvernorat: string, latitude: number, longitude: number): Observable<Delivery[]> {
    return this.http.get<Delivery[]>(`${this.apiUrl}/available`, {
      params: {
        gouvernorat,
        latitude: latitude.toString(),
        longitude: longitude.toString()
      }
    });
  }

  // Accepter une livraison par un chauffeur
  acceptDelivery(deliveryId: number, chauffeurId: number): Observable<Delivery> {
    return this.http.put<Delivery>(`${this.apiUrl}/${deliveryId}/accept/${chauffeurId}`, {});
  }

  // Mettre à jour le statut d'une livraison
  updateDeliveryStatus(deliveryId: number, chauffeurId: number, status: string): Observable<Delivery> {
    return this.http.put<Delivery>(`${this.apiUrl}/${deliveryId}/status/${chauffeurId}`, status);
  }

  // Récupérer toutes les livraisons en attente
  getPendingDeliveries(): Observable<Delivery[]> {
    return this.http.get<Delivery[]>(this.apiUrl);
  }

  // Récupérer les livraisons d'un chauffeur
  getMyDeliveries(chauffeurId: number): Observable<Delivery[]> {
    return this.http.get<Delivery[]>(`${this.apiUrl}/chauffeur/${chauffeurId}/my`);
  }
}

قال ChatGPT:

Ah! This error is classic in Angular:

"Can't bind to 'ngModel' since it isn't a known property of 'input'"

It happens because FormsModule is not imported in the module that declares your component. ngModel is part of Angular Forms, so Angular needs the module to know about it.

✅ Solution

  1. Open your module file, e.g. app.module.ts (or the module where your component is declared).

  2. Import FormsModule:

ts
import { FormsModule } from '@angular/forms';

  1. Add it to the imports array:

ts
@NgModule({
  declarations: [
    AppComponent,
    AjouteLivraisonComponent, // votre composant
    // autres composants
  ],
  imports: [
    BrowserModule,
    FormsModule,   // <-- ajouter FormsModule ici
    HttpClientModule
  ],
  providers: [],
  bootstrap: [AppComponent]
})
export class AppModule { }

After this, [(ngModel)] will work on your inputs and selects.

💡 Tip: If you ever use reactive forms, you would instead import ReactiveFormsModule. But for simple two-way binding like in your form, FormsModule is the right choice.

Ah! This error:

NullInjectorError: No provider for HttpClient!

means Angular doesn’t know how to provide HttpClient. The fix is very simple: you must import HttpClientModule in the module that declares your component.

✅ Solution

  1. Open your module file, e.g., app.module.ts (or the module where your component lives).

  2. Import HttpClientModule:

ts
import { HttpClientModule } from '@angular/common/http';

  1. Add it to the imports array:

ts
@NgModule({
  declarations: [
    AjouteLivraisonComponent,
    // autres composants
  ],
  imports: [
    BrowserModule,
    FormsModule,         // pour ngModel
    HttpClientModule     // <-- ajouter HttpClientModule ici
  ],
  providers: [],
  bootstrap: [AppComponent]
})
export class AppModule { }

✅ After that, Angular will know how to inject HttpClient into your LivraisonService.

💡 Note:

  • Always make sure both FormsModule and HttpClientModule are imported in the module where your component lives.

  • If your component is standalone, you can also add:

imports: [CommonModule, FormsModule, HttpClientModule]

directly in the @Component({ imports: [...] }) metadata.

ts
import { Component, OnInit } from '@angular/core';
import { LivraisonService } from '../../services/livraison.service';
import { Livraison } from '../../models/livraison.model';

@Component({
  selector: 'app-chauffeur-livraisons',
  templateUrl: './chauffeur-livraisons.component.html',
  styleUrls: ['./chauffeur-livraisons.component.css']
})
export class ChauffeurLivraisonsComponent implements OnInit {

  chauffeurId = 1; // remplacer par l'id du chauffeur connecté
  livraisonsDisponibles: Livraison[] = [];
  mesLivraisons: Livraison[] = [];

  constructor(private livraisonService: LivraisonService) {}

  ngOnInit(): void {
    this.chargerLivraisonsDisponibles();
    this.chargerMesLivraisons();
  }

  chargerLivraisonsDisponibles() {
    // Ici on peut mettre un gouvernorat et coordonnées fictives si nécessaire
    this.livraisonService.getAvailableLivraisons('Tunis', 36.8065, 10.1815)
      .subscribe({
        next: (res) => this.livraisonsDisponibles = res,
        error: (err) => console.error(err)
      });
  }

  chargerMesLivraisons() {
    this.livraisonService.getMyLivraisons(this.chauffeurId)
      .subscribe({
        next: (res) => this.mesLivraisons = res,
        error: (err) => console.error(err)
      });
  }

  accepterLivraison(livraisonId: number) {
    this.livraisonService.acceptLivraison(livraisonId, this.chauffeurId)
      .subscribe({
        next: (res) => {
          alert('Livraison acceptée !');
          this.chargerLivraisonsDisponibles();
          this.chargerMesLivraisons();
        },
        error: (err) => {
          console.error(err);
          alert('Erreur lors de l\'acceptation de la livraison.');
        }
      });
  }
}
